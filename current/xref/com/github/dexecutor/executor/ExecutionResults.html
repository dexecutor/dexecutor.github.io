1   package com.github.dexecutor.executor;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   /**
6    * Wrapper class around @ExecutionResult
7    * 
8    * @author Nadeem Mohammad
9    *
10   * @param <T> Type of Node/Task ID
11   * @param <R> Type of Node/Task result
12   */
13  public final class ExecutionResults<T, R> {
14  
15  	private List<ExecutionResult<T, R>> results = new ArrayList<ExecutionResult<T, R>>();
16  	
17  	/**
18  	 * adds {@code result} to existing collection of results
19  	 * 
20  	 * @param result Result to be added to all results
21  	 */
22  	public final void add(final ExecutionResult<T, R> result) {
23  		this.results.add(result);
24  	}
25  	/**
26  	 *  
27  	 * @return the first {@link ExecutionResult in the collection}
28  	 */
29  	public ExecutionResult<T, R> getFirst() {
30  		if (this.results.isEmpty()) {
31  			return null;
32  		} else {
33  			return this.results.iterator().next();
34  		}
35  	}
36  	/**
37  	 * 
38  	 * @return {@code true} If there is any result
39  	 * {@code false} if no result
40  	 */
41  	public boolean hasAnyParentResult() {
42  		if (this.results.isEmpty()) {
43  			return false;
44  		} else {
45  			return true;
46  		}
47  	}
48  	/**
49  	 * 
50  	 * @return all result in the collection
51  	 */
52  	public List<ExecutionResult<T, R>> getAll() {
53  		return new ArrayList<ExecutionResult<T, R>>(this.results);
54  	}
55  
56  	@Override
57  	public String toString() {
58  		return this.results.toString();
59  	}
60  }
