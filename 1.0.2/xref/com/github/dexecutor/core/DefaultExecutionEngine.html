18  package com.github.dexecutor.core;
19  
20  import static com.github.dexecutor.core.support.Preconditions.checkNotNull;
21  
22  import java.util.concurrent.Callable;
23  import java.util.concurrent.CompletionService;
24  import java.util.concurrent.ExecutorCompletionService;
25  import java.util.concurrent.ExecutorService;
26  import java.util.concurrent.Future;
27  
28  import com.github.dexecutor.core.task.ExecutionResult;
29  import com.github.dexecutor.core.task.Task;
30  /**
31   * Default Executor, which internally operates on @ExecutorService
32   * 
33   * @author Nadeem Mohammad
34   *
35   * @param <T>
36   * @param <R>
37   */
38  public final class DefaultExecutionEngine<T extends Comparable<T>, R> implements ExecutionEngine<T, R> {
39  
40  	private final ExecutorService executorService;
41  	private final CompletionService<ExecutionResult<T, R>> completionService;
42  	/**
43  	 * Creates the default instance given @ExecutorService, internally it uses @CompletionService
44  	 * @param executorService
45  	 */
46  	public DefaultExecutionEngine(final ExecutorService executorService) {
47  		checkNotNull(executorService, "Executer Service should not be null");
48  		this.executorService = executorService;
49  		this.completionService = new ExecutorCompletionService<ExecutionResult<T, R>>(executorService);
50  	}
51  
52  	@Override
53  	public Future<ExecutionResult<T, R>> take() throws InterruptedException {
54  		return this.completionService.take();
55  	}
56  
57  	private Callable<ExecutionResult<T, R>> newCallable(final Task<T, R> task) {
58  		return new Callable<ExecutionResult<T,R>>() {
59  
60  			@Override
61  			public ExecutionResult<T, R> call() throws Exception {
62  				R r = task.execute();
63  				return new ExecutionResult<T, R>(task.getId(), r, task.getStatus());
64  			}			
65  		};
66  	}
67  
68  	@Override
69  	public void submit(Task<T, R> task) {
70  		this.completionService.submit(newCallable(task));		
71  	}
72  	
73  	@Override
74  	public String toString() {
75  		return this.executorService.toString();
76  	}
77  }
