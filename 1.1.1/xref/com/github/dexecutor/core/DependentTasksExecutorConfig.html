18  package com.github.dexecutor.core;
19  
20  import static com.github.dexecutor.core.support.Preconditions.checkNotNull;
21  
22  import java.util.concurrent.ExecutorService;
23  
24  import com.github.dexecutor.core.graph.CyclicValidator;
25  import com.github.dexecutor.core.graph.LevelOrderTraversar;
26  import com.github.dexecutor.core.graph.Traversar;
27  import com.github.dexecutor.core.graph.Validator;
28  import com.github.dexecutor.core.task.TaskProvider;
29  
30  /**
31   * <p>Configuration Object for Dexecutor framework. At a minimum it needs {@code ExecutorService} and {@code TaskProvider}, rest are optional and takes default values</p>
32   * <p>This provides way to hook in your own {@code Validator} and {@code Traversar}</p>
33   * 
34   * @author Nadeem Mohammad
35   *
36   * @param <T> Type of Node/Task ID
37   * @param <R> Type of Node/Task result
38   */
39  public class DependentTasksExecutorConfig<T extends Comparable<T>, R> {
40  	
41  	/**
42  	 * Number of threads that should handle the immediate retry.
43  	 */
44  	private int immediateRetryPoolThreadsCount = 1;
45  	
46  	/**
47  	 * Number of threads that should handle the immediate retry.
48  	 */
49  	private int scheduledRetryPoolThreadsCount = 1;
50  
51  	/**
52  	 * executor is the main platform on which tasks are executed
53  	 */
54  	private ExecutionEngine<T, R> executionEngine;
55  
56  	/**
57  	 * When it comes to task execution, task provider would be consulted to provide task objects for execution
58  	 */
59  	private TaskProvider<T, R> taskProvider;
60  	/**
61  	 * Validator for validating the consturcted graph, defaults to detecting Cyclic checks
62  	 */
63  	private Validator<T, R> validator = new CyclicValidator<T, R>();
64  	/**
65  	 * Traversar used to traverse the graph while printing it on a Writer
66  	 */
67  	private Traversar<T, R> traversar = new LevelOrderTraversar<T, R>();
68  	/**
69  	 * Construct the object with mandatory params, rest are optional
70  	 * @param executorService @ExecutorService
71  	 * @param taskProvider @TaskProvider
72  	 */
73  	public DependentTasksExecutorConfig(final ExecutorService executorService, final TaskProvider<T, R> taskProvider) {
74  		checkNotNull(executorService, "Executer Service should not be null");
75  		checkNotNull(taskProvider, "Task Provider should not be null");
76  		this.executionEngine = new DefaultExecutionEngine<>(executorService);
77  		this.taskProvider = taskProvider;
78  	}
79  
80  	/**
81  	 * Construct the object with mandatory params, rest are optional
82  	 * @param executor  @ExecutionEngine
83  	 * @param taskProvider @TaskProvider
84  	 */
85  	public DependentTasksExecutorConfig(final ExecutionEngine<T, R> executor, final TaskProvider<T, R> taskProvider) {
86  		this.executionEngine = executor;
87  		this.taskProvider = taskProvider;
88  	}
89  
90  	void validate() {
91  		checkNotNull(this.executionEngine, "Execution Engine should not be null");
92  		checkNotNull(this.taskProvider, "Task Provider should not be null");
93  		checkNotNull(this.validator, "Validator should not be null");
94  		checkNotNull(this.traversar, "Traversar should not be null");
95  	}
96  
97  	ExecutionEngine<T, R> getExecutorEngine() {
98  		return this.executionEngine;
99  	}
100 
101 	TaskProvider<T, R> getTaskProvider() {
102 		return this.taskProvider;
103 	}
104 
105 	Validator<T, R> getValidator() {
106 		return this.validator;
107 	}
108 	/**
109 	 * change the validator to that of specified
110 	 * @param validator
111 	 */
112 	public void setValidator(final Validator<T, R> validator) {
113 		this.validator = validator;
114 	}
115 	Traversar<T, R> getTraversar() {
116 		return this.traversar;
117 	}
118 	/**
119 	 * Change the traversar to that of specified
120 	 * @param traversar
121 	 */
122 	public void setTraversar(final Traversar<T, R> traversar) {
123 		this.traversar = traversar;
124 	}
125 	/**
126 	 * 
127 	 * @return the immediate retry thread pool count
128 	 */
129 	public int getImmediateRetryPoolThreadsCount() {
130 		return immediateRetryPoolThreadsCount;
131 	}
132 	/**
133 	 * sets the immediate retry thread pool size to that of specified
134 	 * @param immediateRetryPoolThreadsCount
135 	 */
136 	public void setImmediateRetryPoolThreadsCount(int immediateRetryPoolThreadsCount) {
137 		this.immediateRetryPoolThreadsCount = immediateRetryPoolThreadsCount;
138 	}
139 	/**
140 	 * 
141 	 * @return the scheduled retry thread pool size
142 	 */
143 	public int getScheduledRetryPoolThreadsCount() {
144 		return scheduledRetryPoolThreadsCount;
145 	}
146 	/**
147 	 * sets the scheduled thread pool size to that of specified
148 	 * @param scheduledRetryPoolThreadsCount
149 	 */
150 	public void setScheduledRetryPoolThreadsCount(int scheduledRetryPoolThreadsCount) {
151 		this.scheduledRetryPoolThreadsCount = scheduledRetryPoolThreadsCount;
152 	}
153 	
154 }
